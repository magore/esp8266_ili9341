307 void fatfs_ls(char *name)
308 {
309  long p1;
310  UINT s1, s2;
311  int res;
312  FILINFO fno;
313  DIR dirs; /* Directory object */
314  FATFS *fs;
315  char buff[256];
316 
317  if(!name || !*name)
318  {
319  strcpy(buff,".");
320  }
321  else
322  {
323  strcpy(buff,name);
324  }
325  printf("Listing:[%s]\n",buff);
326 
327  res = f_opendir(&dirs, buff);
328  if (res) { put_rc(res); return; }
329  p1 = s1 = s2 = 0;
330  while(1)
331  {
332  res = f_readdir(&dirs, &fno);
333  if ((res != FR_OK) || !fno.fname[0]) break;
334  if (fno.fattrib & AM_DIR)
335  {
336  s2++;
337  }
338  else
339  {
340  s1++; p1 += fno.fsize;
341  }
342  fatfs_filinfo_list(&fno);
343 #ifdef ESP8266
344  optimistic_yield(1000);
345  wdt_reset();
346 #endif
347  }
348  printf("%4u File(s),%10lu bytes total\n%4u Dir(s)", s1, p1, s2);
349  if (f_getfree(buff, (DWORD*)&p1, &fs) == FR_OK)
350  printf(", %10luK bytes free\n", p1 * fs->csize / 2);
351 }
