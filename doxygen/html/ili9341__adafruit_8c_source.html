473 MEMSPACE
474 void tft_fillTriangle ( window *win, int16_t x0, int16_t y0,
475 int16_t x1, int16_t y1,
476 int16_t x2, int16_t y2, uint16_t color)
477 {
478 
479  int16_t a, b, y, last;
480 
481 // Sort coordinates by Y order (y2 >= y1 >= y0)
482  if (y0 > y1)
483  {
484  SWAP(y0, y1); SWAP(x0, x1);
485  }
486  if (y1 > y2)
487  {
488  SWAP(y2, y1); SWAP(x2, x1);
489  }
490  if (y0 > y1)
491  {
492  SWAP(y0, y1); SWAP(x0, x1);
493  }
494 
495  if(y0 == y2) // Handle awkward all-on-same-line case as its own thing
496  {
497  a = b = x0;
498  if(x1 < a) a = x1;
499  else if(x1 > b) b = x1;
500  if(x2 < a) a = x2;
501  else if(x2 > b) b = x2;
502  tft_drawFastHLine(win,a, y0, b-a+1, color);
503  return;
504  }
505 
506  int16_t
507  dx01 = x1 - x0,
508  dy01 = y1 - y0,
509  dx02 = x2 - x0,
510  dy02 = y2 - y0,
511  dx12 = x2 - x1,
512  dy12 = y2 - y1,
513  sa = 0,
514  sb = 0;
515 
516 // For upper part of triangle, find scanline crossings for segments
517 // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
518 // is included here (and second loop will be skipped, avoiding a /0
519 // error there), otherwise scanline y1 is skipped here and handled
520 // in the second loop...which also avoids a /0 error here if y0=y1
521 // (flat-topped triangle).
522  if(y1 == y2) last = y1; // Include y1 scanline
523  else last = y1-1; // Skip it
524 
525  for(y=y0; y<=last; y++)
526  {
527  a = x0 + sa / dy01;
528  b = x0 + sb / dy02;
529  sa += dx01;
530  sb += dx02;
531 /* longhand:
532 a = x0 + (x1 - x0) * (y - y0) / (y1 - y0);
533 b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
534  */
535  if(a > b) SWAP(a,b);
536  tft_drawFastHLine(win,a, y, b-a+1, color);
537  }
538 
539 // For lower part of triangle, find scanline crossings for segments
540 // 0-2 and 1-2. This loop is skipped if y1=y2.
541  sa = dx12 * (y - y1);
542  sb = dx02 * (y - y0);
543  for(; y<=y2; y++)
544  {
545  a = x1 + sa / dy12;
546  b = x0 + sb / dy02;
547  sa += dx12;
548  sb += dx02;
549 /* longhand:
550 a = x1 + (x2 - x1) * (y - y1) / (y2 - y1);
551 b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
552  */
553  if(a > b) SWAP(a,b);
554  tft_drawFastHLine(win,a, y, b-a+1, color);
555  }
556 }
