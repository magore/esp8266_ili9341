207  Yavg /= XPT2046_SAMPLES;

222 
223  T = XPT2046_read_filtered((uint16_t *)&X, (uint16_t *)&Y);
224 
225  // Key debounce state machine
226  switch(xpt2046.state)
227  {
228  // Full init
229  case 0:
230  xpt2046.state = 1;
231  xpt2046.ms = 0;
232  xpt2046.count = 0;
233  xpt2046.head = 0;
234  xpt2046.tail = 0;
235  // Fall through to state 1
236  case 1:
237  if(T) // key is down
238  {
239  // wait for key down debounce time before taking a sample
240  if(++xpt2046.ms < XPT2046_DEBOUNCE)
241  break;
242 
243  if(xpt2046.count < XPT2046_EVENTS )
244  {
245  xpt2046.XQ[xpt2046.head] = X;
246  xpt2046.YQ[xpt2046.head] = Y;
247  if(++xpt2046.head >= XPT2046_EVENTS)
248  xpt2046.head = 0;
249  xpt2046.count++;
250  xpt2046.state = 2;
251  xpt2046.ms = 0;
252  }
253  else
254  {
255  xpt2046.ms = 0;
256  }
257  } // if (T)
258  else // Restart timer and wait for touch debounce time
259  {
260  xpt2046.ms = 0;
261  }
262  break;
263 
264  // Wait for touch release debounce time
265  case 2:
266  // Released ?
267  if(!T)

269  // Debounce release time - valid key depress cycle done
