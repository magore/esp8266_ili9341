652 {
653  time_t epoch;
654  int32_t offset;
655  int isdst;
656 
657 // struct tm
658 // {
659 // int tm_sec; /*< Seconds. [0-60] (1 leap second) */
660 // int tm_min; /*< Minutes. [0-59] */
661 // int tm_hour; /*< Hours. [0-23] */
662 // int tm_mday; /*< Day. [1-31] */
663 // int tm_mon; /*< Month. [0-11] */
664 // int tm_year; /*< Year - 1900. */
665 // int tm_wday; /*< Day of week. [0-6] */
666 // int tm_yday; /*< Days in year.[0-365] */
667 // int tm_isdst; /*< DST. [-1/0/1] */
668 // };
669 
670  // Normalize t->tm_seconds
671  while(t->tm_sec >= 60)
672  {
673  ++t->tm_min;
674  t->tm_sec -= 60;
675  }
676  while(t->tm_sec < 0)
677  {
678  --t->tm_min;
679  t->tm_sec += 60;
680  }
681 
682  // Normalize t->tm_miniutes
683  while(t->tm_min >= 60)
684  {
685  ++t->tm_hour;
686  t->tm_min -= 60;
687  }
688  while(t->tm_min < 0)
689  {
690  --t->tm_hour;
691  t->tm_min += 60;
692  }
693 
694  // Normalize t->tm_hours
695  while(t->tm_hour >= 24)
696  {
697  ++t->tm_mday;
698  t->tm_hour -= 24;
699  }
700  while(t->tm_hour < 0)
701  {
702  --t->tm_mday;
703  t->tm_hour += 24;
704  }
705 
706  // Normalize t->tm_months
707  while(t->tm_mon >= 12)
708  {
709  ++t->tm_year;
710  t->tm_mon -= 12;
711  }
712  while(t->tm_mon < 0)
713  {
714  --t->tm_year;
715  t->tm_mon += 12;
716  }
717 
718  // Normalize t->tm_mday
719  // t->tm_mday is 1 based
720  while(t->tm_mday > Days_Per_Month(t->tm_mon,t->tm_year) )
721  {
722  // subtract days in current t->tm_month
723  t->tm_mday -= Days_Per_Month(t->tm_mon,t->tm_year);
724  // Keep month normalized
725  if(++t->tm_mon >= 12)
726  {
727  t->tm_mon -= 12;
728  ++t->tm_year;
729  }
730  }
731

733  while(t->tm_mday < 1)
734  {
735  // Keep month normalized
736  if(--t->tm_mon < 0)
737  {
738  t->tm_mon += 12;
739  --t->tm_year;
740  }
741  // add days in previous mount
742  t->tm_mday += Days_Per_Month(t->tm_mon,t->tm_year);
743  }
744 
745 
746  // We can now set the remain values by converting to EPOCH and back again
747  // convert to EPOCH based seconds
748  epoch = tm2epoch( t );
749 #ifdef TIME_DEBUG
750  printf("%10lu - epoch-1\n",epoch);
751 #endif
752  // Normaize to local timezone with DST
753  offset = 0L;
754  isdst = 0;
755  if(normalize_to_timezone)
756  {
757  tz_t tz;
758 
759  gettimezone(&tz);
760  offset = tz.tz_minuteswest * 60L;
761  if(is_dst(epoch))
762  {
763  offset -= 3600L;
764  isdst = 1;
765  }
766  }
767 #ifdef TIME_DEBUG

769 #endif
770

775

778 #endif
779  return( epoch );
780 }
