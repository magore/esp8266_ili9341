204 static void hspi_writeFIFO(uint8_t *write_data, int bytes)
205 {
206  uint8_t word_ind = 0;
207 
208  if(bytes > HSPI_FIFO_SIZE) // TODO set error status
209  return;
210 
211  hspi_setBits(bytes); // Update FIFO with number of bits we will send
212 
213 // First do a fast write with 32 bit chunks at a time
214  while(bytes >= 4)
215  {
216 // Cast both source and destination to 4 byte word pointers
217  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = \
218  ((uint32_t *)write_data) [word_ind];
219 
220  bytes -= 4;
221  word_ind++;
222  }
223 
224 // Next write remaining bytes (if any) to FIFO as a single word
225 // Note: We follow the good practice of avoiding access past the end of an array.
226 // (ie. protected memory, PIC using retw from flash ... could crash some CPU's)
227  if(bytes) // Valid counts are now: 3,2,1 bytes
228  {
229  uint32_t last_word = 0; // Last word to send
230  uint16_t byte_ind = word_ind << 2; // Convert to Byte index
231 
232 // Working MSB to LSB allows assigment to LSB without shifting
233 // Compiler can optimize powers of 8 into byte swaps, etc... (on some CPU's)
234  while(bytes--) // index is now 2,1,0 matching required storage offset
235  {
236  last_word <<= 8;
237 // 2,1,0
238  last_word |= ((uint32_t) 0xffU & write_data[byte_ind + bytes ]);
239  }
240 // Send last partial word
241  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = last_word;
242  }
243 }
