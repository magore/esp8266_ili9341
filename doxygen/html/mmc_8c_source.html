194 BYTE send_cmd (
195 BYTE cmd, /*< Command index */
196 DWORD arg /*< Argument */
197 )
198 {
199  BYTE n, res;
200 
201  if (cmd & 0x80) /* ACMD<n> is the command sequense of CMD55-CMD<n> */
202  {
203  cmd &= 0x7F;
204  res = send_cmd(CMD55, 0);
205  if (res > 1) return res;
206  }
207 
208 /* Select the card and wait for ready except to stop multiple block read */
209  if (cmd != CMD12)
210  {
211  mmc_deselect();
212  if (!mmc_select()) return 0xFF;
213  }
214 
215 /* Send command packet */
216  mmc_spi_TX(0x40 | cmd); /* Start + Command index */
217  mmc_spi_TX((BYTE)(arg >> 24)); /* Argument[31..24] */
218  mmc_spi_TX((BYTE)(arg >> 16)); /* Argument[23..16] */
219  mmc_spi_TX((BYTE)(arg >> 8)); /* Argument[15..8] */
220  mmc_spi_TX((BYTE)arg); /* Argument[7..0] */
221  n = 0x01; /* Dummy CRC + Stop */
222  if (cmd == CMD0) n = 0x95; /* Valid CRC for CMD0(0) + Stop */
223  if (cmd == CMD8) n = 0x87; /* Valid CRC for CMD8(0x1AA) Stop */
224 
225  mmc_spi_TX(n);
226 
227 /* Receive command response */
228  if (cmd == CMD12) mmc_spi_TX(0xFF); /* Skip a stuff byte when stop reading */
229  n = 10; /* Wait for a valid response in timeout of 10 attempts */
230  do
231  {
232  res = mmc_spi_TXRX(0xFF);
233  }
234  while ((res & 0x80) && --n);
235 
236  return res; /* Return with the response value */
237 }

246 DSTATUS mmc_disk_initialize (
247 BYTE pdrv /*< Physical drive nmuber (0) */
248 )
249 {
250  BYTE n, cmd, ty, ocr[4];
251 
252  if (pdrv) return STA_NOINIT; /* Supports only single drive */
253  mmc_power_off(); /* Turn off the sock et power to reset the card */
254  if (Stat & STA_NODISK) return Stat; /* No card in the socket */
255  mmc_power_on(); /* Turn on the socket power */
256  mmc_slow();
257  for (n = 10; n; n--) mmc_spi_TX(0xFF); /* 80 dummy clocks */
258 
259  ty = 0;
260  if (send_cmd(CMD0, 0) == 1) /* Enter Idle state */
261  {
262  mmc_set_ms_timeout(250); /* Initialization timeout of 1000 msec */
263  if (send_cmd(CMD8, 0x1AA) == 1) /* SDv2? */
264  {
265  for (n = 0; n < 4; n++)
266  {
267  ocr[n] = mmc_spi_TXRX(0xFF); /* Get trailing return value of R7 resp */
268  }
269  if (ocr[2] == 0x01 && ocr[3] == 0xAA) /* The card can work at vdd range of 2.7-3.6V */
270  {
271 /* Wait for leaving idle state (ACMD41 with HCS bit) */
272  while (!mmc_test_timeout() && send_cmd(ACMD41, 1UL << 30))
273  {
274  }
275 /* Check CCS bit in the OCR */
276  if (!mmc_test_timeout() && send_cmd(CMD58, 0) == 0)
277  {
278  for (n = 0; n < 4; n++) ocr[n] = mmc_spi_TXRX(0xFF);
279 /* SDv2 */
280  ty = (ocr[0] & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2;
281  }
282  }
283  }
284  else
285  {
286 /* SDv1 or M
287 MCv3 */
288  if (send_cmd(ACMD41, 0) <= 1)
289  {
290  ty = CT_SD1; cmd = ACMD41; /* SDv1 */
291  }
292  else
293  {
294  ty = CT_MMC; cmd = CMD1; /* MMCv3 */
295  }
296 /* Wait for leaving idle state */
297  while (!mmc_test_timeout() && send_cmd(cmd, 0));
298 /* Set R/W block length to 512 */
299  if (mmc_test_timeout() || send_cmd(CMD16, 512) != 0)
300  ty = 0;
301  }
302  }
303 
304  CardType = ty;
305  mmc_deselect();
306 
307  if (ty) /* Initialization succeded */
308  {
309  Stat &= ~STA_NOINIT; /* Clear STA_NOINIT */
310  mmc_fast();
311  } /* Initialization failed */
312  else
313  {
314  mmc_power_off();
315  }
316 
317  if (Stat & STA_NODISK) return Stat; /* No card in the socket */
318 
319  return Stat;
320 }

436 BYTE cmd, /* Control code */

443

445 
446  res = RES_ERROR;
447 
448  if (Stat & STA_NOINIT) return RES_NOTRDY;
449 
450  switch (cmd)
451  {
452  case CTRL_SYNC : /* Make sure that no pending write process. Do not remove this or written sector might not left updated. */
453  if (mmc_select()) res = RES_OK;
454  break;
455 
456  case GET_SECTOR_COUNT : /* Get number of sectors on the disk (DWORD) */
457  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16))
458  {
459  if ((csd[0] >> 6) == 1) /* SDC ver 2.00 */
460  {
461  csize = csd[9] + ((WORD)csd[8] << 8) + ((DWORD)(csd[7] & 63) << 16) + 1;
462  *(DWORD*)buff = csize << 10;
463  } /* SDC ver 1.XX or M MC*/
464  else
465  {
466  n = (csd[5] & 15) + ((csd[10] & 128) >> 7) + ((csd[9] & 3) << 1) + 2;
467  csize = (csd[8] >> 6) + ((WORD)csd[7] << 2) + ((WORD)(csd[6] & 3) << 10) + 1;
468  *(DWORD*)buff = csize << (n - 9);
469  }
470  res = RES_OK;
471  }
472  break;
473 
474  case GET_BLOCK_SIZE : /* Get erase block size in unit of sector (DWORD) */
475  if (CardType & CT_SD2) /* SDv2? */
476  {
477  if (send_cmd(ACMD13, 0) == 0) /* Read SD status */
478  {
479  mmc_spi_TX(0xFF);
480  if (rcvr_datablock(csd, 16)) /* Read partial block */
481  {
482 /* Purge trailing data */
483  for (n = 64 - 16; n; n--) mmc_spi_TX(0xFF);
484  *(DWORD*)buff = 16UL << (csd[10] >> 4);
485  res = RES_OK;
486  }
487  }
488  } /* SDv1 or MMCv3 */
489  else
490  {
491 /* Read CSD */
492  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16))
493  {
494  if (CardType & CT_SD1) /* SDv1 */
495  {
496  *(DWORD*)buff = (((csd[10] & 63) << 1) + ((WORD)(csd[11] & 128) >> 7) + 1) << ((csd[13] >> 6) - 1);
497  } /* MMCv3 */
498  else
499  {
500  *(DWORD*)buff = ((WORD)((csd[10] & 124) >> 2) + 1) * (((csd[11] & 3) << 3) + ((csd[11] & 224) >> 5) + 1);
501  }
502  res = RES_OK;
503  }
504  }
505  break;
506

508

510  *ptr = CardType;
511  res = RES_OK;
512  break;
513 
514  case MMC_GET_CSD : /* Receive CSD as a data block (16 bytes) */
515  if (send_cmd(CMD9, 0) == 0 /* READ_CSD */

530  res = RES_OK;
531  }
532  break;

537  mmc_spi_TX(0xFF);
