287 static
288 BYTE send_cmd (
289 BYTE cmd, /*< Command index */
290 DWORD arg /*< Argument */
291 )
292 {
293  BYTE n, res;
294 
295  if (cmd & 0x80) /* ACMD<n> is the command sequense of CMD55-CMD<n> */
296  {
297  cmd &= 0x7F;
298  res = send_cmd(CMD55, 0);
299  if (res > 1) return res;
300  }
301 
302 /* Select the card and wait for ready except to stop multiple block read */
303  if (cmd != CMD12)
304  {
305  deselect();
306  if (!select()) return 0xFF;
307  }
308 
309 /* Send command packet */
310  xchg_spi(0x40 | cmd); /* Start + Command index */
311  xchg_spi((BYTE)(arg >> 24)); /* Argument[31..24] */
312  xchg_spi((BYTE)(arg >> 16)); /* Argument[23..16] */
313  xchg_spi((BYTE)(arg >> 8)); /* Argument[15..8] */
314  xchg_spi((BYTE)arg); /* Argument[7..0] */
315  n = 0x01; /* Dummy CRC + Stop */
316  if (cmd == CMD0) n = 0x95; /* Valid CRC for CMD0(0) + Stop */
317  if (cmd == CMD8) n = 0x87; /* Valid CRC for CMD8(0x1AA) Stop */
318 
319  xchg_spi(n);
320 
321 /* Receive command response */
322  if (cmd == CMD12) xchg_spi(0xFF); /* Skip a stuff byte when stop reading */
323  n = 10; /* Wait for a valid response in timeout of 10 attempts */
324  do
325  {
326  res = xchg_spi(0xFF);
327  }
328  while ((res & 0x80) && --n);
329 
330  return res; /* Return with the response value */
331 }

340 DSTATUS mmc_disk_initialize (void)
341 {
342  BYTE n, cmd, ty, ocr[4];
343 
344  //for (Timer1 = 10; Timer1; ) ; /* Wait for 100ms */
345  if (Stat & STA_NODISK) return Stat; /* No card in the socket */
346 
347  FCLK_SLOW();
348 
349  for (n = 10; n; n--) xchg_spi(0xFF); /* 80 dummy clocks */
350 
351  ty = 0;
352  if (send_cmd(CMD0, 0) == 1) /* Enter Idle state */
353  {
354  //Timer1=100; /* Initialization timeout of 1000 msec */
355  mmc_set_ms_timeout(1000); /* Initialization timeout of 1000 msec */
356  if (send_cmd(CMD8, 0x1AA) == 1) /* SDv2? */
357  {
358  for (n = 0; n < 4; n++)
359  ocr[n] = xchg_spi(0xFF); /* Get trailing return value of R7 resp */
360  if (ocr[2] == 0x01 && ocr[3] == 0xAA) /* The card can work at vdd range of 2.7-3.6V */
361  {
362  /* Wait for leaving idle state (ACMD41 with HCS bit) */
363  //while (Timer1 && send_cmd(ACMD41, 1UL << 30))
364  while (!mmc_test_timeout() && send_cmd(ACMD41, 1UL << 30))
365  ;
366  /* Check CCS bit in the OCR */
367  //if (Timer1 && send_cmd(CMD58, 0) == 0)
368  if (!mmc_test_timeout() && send_cmd(CMD58, 0) == 0)
369  {
370  for (n = 0; n < 4; n++)
371  ocr[n] = xchg_spi(0xFF);
372  /* Check if the card is SDv2 */
373  ty = (ocr[0] & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2;
374  }
375  }
376  }
377  else
378  { /* SDv1 or MMCv3 */
379  if (send_cmd(ACMD41, 0) <= 1)
380  {
381  ty = CT_SD1; cmd = ACMD41; /* SDv1 */
382  }
383  else
384  {
385  ty = CT_MMC; cmd = CMD1; /* MMCv3 */
386  }
387  /* Wait for leaving idle state */
388  //while (Timer1 && send_cmd(cmd, 0))
389  while (!mmc_test_timeout() && send_cmd(cmd, 0))
390  ;
391  /* Set R/W block length to 512 */
392  //if (!Timer1 || send_cmd(CMD16, 512) != 0)
393  if (mmc_test_timeout() || send_cmd(CMD16, 512) != 0)
394  ty = 0;
395  }
396  }
397 
398  CardType = ty;
399  deselect();
400 
401  if (ty) /* Initialization succeded */
402  {
403  Stat &= ~STA_NOINIT; /* Clear STA_NOINIT */
404  FCLK_FAST();
405  } /* Initialization failed */
406  else
407  {
408  power_off();
409  }
410 
411  return Stat;
412 }

432 DRESULT mmc_disk_read (
433 BYTE *buff, /*< Pointer to the data buffer to store read data */
434 DWORD sector, /*< Start sector number (LBA) */
435 UINT count /*< Sector count (1..128) */
436 )
437 {
438  BYTE cmd;
439 
440  if (!count)
441  {
442  deselect();
443  return RES_PARERR;
444  }
445  if (Stat & STA_NOINIT)
446  {
447  deselect();
448  return RES_NOTRDY;
449  }
450 
451  if (!(CardType & CT_BLOCK)) sector *= 512; /* Convert to byte address if needed */
452 
453  cmd = count > 1 ? CMD18 : CMD17; /* READ_MULTIPLE_BLOCK : READ_SINGLE_BLOCK */
454  if (send_cmd(cmd, sector) == 0)
455  {
456  do
457  {
458  if (!rcvr_datablock(buff, 512))
459  break;
460  buff += 512;
461  } while (--count);
462  if (cmd == CMD18) send_cmd(CMD12, 0); /* STOP_TRANSMISSION */
463  }
464  deselect();
465 
466  return count ? RES_ERROR : RES_OK;
467 }

533 MEMSPACE
534 DRESULT mmc_disk_ioctl (
535  BYTE cmd, /* Control code */
536  void *buff /* Buffer to send/receive control data */
537 )
538 {
539  DRESULT res = RES_NOTRDY;
540  BYTE n, csd[16], *ptr = buff;
541  DWORD *dp, st, ed, csize;
542 #if _USE_ISDIO
543  SDIO_CTRL *sdi;
544  BYTE rc, *bp;
545  UINT dc;
546 #endif
547 
548  if (Stat & STA_NOINIT)
549  {
550  return RES_NOTRDY;
551  }
552 
553  res = RES_ERROR;
554  switch (cmd) {
555  case CTRL_SYNC : /* Make sure that no pending write process. Do not remove this or written sector might not left updated. */
556  if (select()) res = RES_OK;
557 //MG
558 deselect();
559  break;
560 
561  case GET_SECTOR_COUNT : /* Get number of sectors on the disk (DWORD) */
562  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) {
563  if ((csd[0] >> 6) == 1) { /* SDC ver 2.00 */
564  csize = csd[9] + ((WORD)csd[8] << 8) + ((DWORD)(csd[7] & 63) << 16) + 1;
565  *(DWORD*)buff = csize << 10;
566  } else { /* SDC ver 1.XX or MMC*/
567  n = (csd[5] & 15) + ((csd[10] & 128) >> 7) + ((csd[9] & 3) << 1) + 2;
568  csize = (csd[8] >> 6) + ((WORD)csd[7] << 2) + ((WORD)(csd[6] & 3) << 10) + 1;
569  *(DWORD*)buff = csize << (n - 9);
570  }
571  res = RES_OK;
572  }
573  deselect();
574  break;
575 
576  case GET_BLOCK_SIZE : /* Get erase block size in unit of sector (DWORD) */
577  if (CardType & CT_SD2) { /* SDv2? */
578  if (send_cmd(ACMD13, 0) == 0) { /* Read SD status */
579  xchg_spi(0xFF);
580  if (rcvr_datablock(csd, 16)) { /* Read partial block */
581  for (n = 64 - 16; n; n--) xchg_spi(0xFF); /* Purge trailing data */
582  *(DWORD*)buff = 16UL << (csd[10] >> 4);
583  res = RES_OK;
584  }
585  }
586  } else { /* SDv1 or MMCv3 */
587  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) { /* Read CSD */
588  if (CardType & CT_SD1) { /* SDv1 */
589  *(DWORD*)buff = (((csd[10] & 63) << 1) + ((WORD)(csd[11] & 128) >> 7) + 1) << ((csd[13] >> 6) - 1);
590  } else { /* MMCv3 */
591  *(DWORD*)buff = ((WORD)((csd[10] & 124) >> 2) + 1) * (((csd[11] & 3) << 3) + ((csd[11] & 224) >> 5) + 1);
592  }
593  res = RES_OK;
594  }
595  }
596  deselect();
597  break;
598 
599  case CTRL_TRIM: /* Erase a block of sectors (used when _USE_TRIM in ffconf.h is 1) */
600  if (!(CardType & CT_SDC)) break; /* Check if the card is SDC */
601  if (mmc_disk_ioctl(MMC_GET_CSD, csd)) break; /* Get CSD */
602  if (!(csd[0] >> 6) && !(csd[10] & 0x40)) break; /* Check if sector erase can be applied to the card */
603  dp = buff; st = dp[0]; ed = dp[1]; /* Load sector block */
604  if (!(CardType & CT_BLOCK)) {
605  st *= 512; ed *= 512;
606  }
607  if (send_cmd(CMD32, st) == 0 && send_cmd(CMD33, ed) == 0 && send_cmd(CMD38, 0) == 0 && wait_ready(30000)) /* Erase sector block */
608  res = RES_OK; /* FatFs does not check result of this command */
609 //MG
610 deselect();
611  break;
612 
613  /* Following commands are never used by FatFs module */
614 
615  case MMC_GET_TYPE : /* Get card type flags (1 byte) */
616  *ptr = CardType;
617  res = RES_OK;
618  break;
619 
620  case MMC_GET_CSD : /* Receive CSD as a data block (16 bytes) */
621  if (send_cmd(CMD9, 0) == 0 && rcvr_datablock(ptr, 16)) /* READ_CSD */
622  res = RES_OK;
623  deselect();
624  break;
625 
626  case MMC_GET_CID : /* Receive CID as a data block (16 bytes) */
627  if (send_cmd(CMD10, 0) == 0 && rcvr_datablock(ptr, 16)) /* READ_CID */
628 
629  res = RES_OK;
630  deselect();
631  break;
632 
633  case MMC_GET_OCR : /* Receive OCR as an R3 resp (4 bytes) */
634  if (send_cmd(CMD58, 0) == 0) { /* READ_OCR */
635  for (n = 4; n; n--) *ptr++ = xchg_spi(0xFF);
636  res = RES_OK;
637  }
638  deselect();
639  break;
640 
641  case MMC_GET_SDSTAT : /* Receive SD statsu as a data block (64 bytes) */
642  if (send_cmd(ACMD13, 0) == 0) { /* SD_STATUS */
643  xchg_spi(0xFF);
644  if (rcvr_datablock(ptr, 64)) res = RES_OK;
645  }
646  deselect();
647  break;
648 
649  case CTRL_POWER_OFF : /* Power off */
650  power_off();
651  Stat |= STA_NOINIT;
652  res = RES_OK;
653  break;
654 #if _USE_ISDIO
655  case ISDIO_READ:
656  sdi = buff;
657  if (send_cmd(CMD48, 0x80000000 | (DWORD)sdi->func << 28 | (DWORD)sdi->addr << 9 | ((sdi->ndata - 1) & 0x1FF)) == 0) {
658  //for (Timer1 = 100; (rc = xchg_spi(0xFF)) == 0xFF && Timer1; ) ;
659  mmc_set_ms_timeout(1000);
660  while( (rc = xchg_spi(0xFF)) == 0xFF && !mmc_test_timeout() )
661  ;
662  if (rc == 0xFE) {
663  for (bp = sdi->data, dc = sdi->ndata; dc; dc--) *bp++ = xchg_spi(0xFF);
664  for (dc = 514 - sdi->ndata; dc; dc--) xchg_spi(0xFF);
665  res = RES_OK;
666  }
667  }
668  deselect();
669  break;
670 
671  case ISDIO_WRITE:
672  sdi = buff;
673  if (send_cmd(CMD49, 0x80000000 | (DWORD)sdi->func << 28 | (DWORD)sdi->addr << 9 | ((sdi->ndata - 1) & 0x1FF)) == 0) {
674  xchg_spi(0xFF); xchg_spi(0xFE);
675  for (bp = sdi->data, dc = sdi->ndata; dc; dc--) xchg_spi(*bp++);
676  for (dc = 514 - sdi->ndata; dc; dc--) xchg_spi(0xFF);
677  if ((xchg_spi(0xFF) & 0x1F) == 0x05) res = RES_OK;
678  }
679  deselect();
680  break;
681 
682  case ISDIO_MRITE:
683  sdi = buff;
684  if (send_cmd(CMD49, 0x84000000 | (DWORD)sdi->func << 28 | (DWORD)sdi->addr << 9 | sdi->ndata >> 8) == 0) {
685  xchg_spi(0xFF); xchg_spi(0xFE);
686  xchg_spi(sdi->ndata);
687  for (dc = 513; dc; dc--) xchg_spi(0xFF);
688  if ((xchg_spi(0xFF) & 0x1F) == 0x05) res = RES_OK;
689  }
690  deselect();
691  break;
692 #endif

694  res = RES_PARERR;
