492 void uart0_gets_add(uint8_t data)
493 {
494  int ind;
495  if(!uart0_gets_size || !uart0_gets_receive_queue || !uart0_gets_line)
496  return;
497 
498  // note: we buffer only the most receint line received
499  if(data == '\r')
500  {
501  // transfer current queue to uart0_gets_line buffer
502  ind = 0;
503  while(!queue_empty(uart0_gets_receive_queue) && ind < uart0_gets_size)
504  {
505  uart0_gets_line[ind++] = queue_popc(uart0_gets_receive_queue);
506  }
507  uart0_gets_line[ind++] = 0;
508  uart0_gets_ready = 1;
509  }
510  else
511  {
512  // filter input data
513  if(data == '\t' || (data >= ' ' && data <= 0x7e))
514  {
515  if(queue_space(uart0_gets_receive_queue) )
516  queue_pushc(uart0_gets_receive_queue, data);
517  }
518  }

587 void uart_config(uint8 uart_no, uint32_t baud, uint8_t data_bits, uint8_t stop_bits, uint8_t parity)
588 {
589  if (uart_no == UART0)
590  {
591  PIN_PULLUP_DIS(PERIPHS_IO_MUX_U0TXD_U);
592  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0TXD_U, FUNC_U0TXD);
593  PIN_PULLUP_DIS (PERIPHS_IO_MUX_U0RXD_U);
594  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0RXD_U, FUNC_U0RXD);
595  }
596  else
597  {
598 // GPIO2
599  PIN_PULLUP_DIS(PERIPHS_IO_MUX_GPIO2_U);
600  PIN_FUNC_SELECT(PERIPHS_IO_MUX_GPIO2_U, FUNC_U1TXD_BK);
601  }
602 
603  uart_div_modify(uart_no, UART_CLK_FREQ / baud);
604 
605 // Data bits 5..8
606  data_bits = data_bits - 5;
607 
608 // STOP BITS 1..2
609  if(stop_bits == 2)
610  stop_bits = TWO_STOP_BIT;
611  else
612  stop_bits = ONE_STOP_BIT;
613 
614 // PARITY NO_PARITY,ODD_PARITY,EVEN_PARITY
615 
616  WRITE_PERI_REG(UART_CONF0(uart_no),
617  ((data_bits & UART_BIT_NUM) << UART_BIT_NUM_S) |
618  ((stop_bits & UART_STOP_BIT_NUM) << UART_STOP_BIT_NUM_S) |
619  parity);
620 
621 //clear rx and tx fifo,not ready
622 //RESET FIFO
623  tx_fifo_flush(uart_no);
624  rx_fifo_flush(uart_no);
625 
626  if (uart_no == UART0)
627  {
628  WRITE_PERI_REG(UART_CONF1(uart_no),
629  (( 2 & UART_RX_TOUT_THRHD) << UART_RX_TOUT_THRHD_S) | UART_RX_TOUT_EN |
630  ((16 & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S) |
631  ((64 & UART_TXFIFO_EMPTY_THRHD) << UART_TXFIFO_EMPTY_THRHD_S) );
632  WRITE_PERI_REG(UART_INT_CLR(uart_no), 0xffff);
633  WRITE_PERI_REG(UART_INT_ENA(uart_no), UART_RXFIFO_TOUT_INT_ENA | UART_RXFIFO_FULL_INT_ENA);
634  }
635 }
