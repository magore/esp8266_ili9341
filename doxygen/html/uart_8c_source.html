497 void uart0_gets_add(uint8_t data)
498 {
499  int ind;
500  if(!uart0_gets_size || !uart0_gets_receive_queue || !uart0_gets_line)
501  return;
502 
503  // note: we buffer only the most receint line received
504  if(data == '\r')
505  {
506  // transfer current queue to uart0_gets_line buffer
507  ind = 0;
508  while(!queue_empty(uart0_gets_receive_queue) && ind < uart0_gets_size)
509  {
510  uart0_gets_line[ind++] = queue_popc(uart0_gets_receive_queue);
511  }
512  uart0_gets_line[ind++] = 0;
513  uart0_gets_ready = 1;
514  }
515  else
516  {
517  // filter input data
518  if(data == '\t' || (data >= ' ' && data <= 0x7e))
519  {
520  if(queue_space(uart0_gets_receive_queue) )
521  queue_pushc(uart0_gets_receive_queue, data);
522  }
523  }

592 void uart_config(uint8 uart_no, uint32_t baud, uint8_t data_bits, uint8_t stop_bits, uint8_t parity)
593 {
594  if (uart_no == UART0)
595  {
596  PIN_PULLUP_DIS(PERIPHS_IO_MUX_U0TXD_U);
597  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0TXD_U, FUNC_U0TXD);
598  PIN_PULLUP_DIS (PERIPHS_IO_MUX_U0RXD_U);
599  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0RXD_U, FUNC_U0RXD);
600  }
601  else
602  {
603 // GPIO2
604  PIN_PULLUP_DIS(PERIPHS_IO_MUX_GPIO2_U);
605  PIN_FUNC_SELECT(PERIPHS_IO_MUX_GPIO2_U, FUNC_U1TXD_BK);
606  }
607 
608  uart_div_modify(uart_no, UART_CLK_FREQ / baud);
609 
610 // Data bits 5..8
611  data_bits = data_bits - 5;
612 
613 // STOP BITS 1..2
614  if(stop_bits == 2)
615  stop_bits = TWO_STOP_BIT;
616  else
617  stop_bits = ONE_STOP_BIT;
618 
619 // PARITY NO_PARITY,ODD_PARITY,EVEN_PARITY
620 
621  WRITE_PERI_REG(UART_CONF0(uart_no),
622  ((data_bits & UART_BIT_NUM) << UART_BIT_NUM_S) |
623  ((stop_bits & UART_STOP_BIT_NUM) << UART_STOP_BIT_NUM_S) |
624  parity);
625 
626 //clear rx and tx fifo,not ready
627 //RESET FIFO
628  tx_fifo_flush(uart_no);
629  rx_fifo_flush(uart_no);
630 
631  if (uart_no == UART0)
632  {
633  WRITE_PERI_REG(UART_CONF1(uart_no),
634  (( 2 & UART_RX_TOUT_THRHD) << UART_RX_TOUT_THRHD_S) | UART_RX_TOUT_EN |
635  ((16 & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S) |
636  ((64 & UART_TXFIFO_EMPTY_THRHD) << UART_TXFIFO_EMPTY_THRHD_S) );
637  WRITE_PERI_REG(UART_INT_CLR(uart_no), 0xffff);
638  WRITE_PERI_REG(UART_INT_ENA(uart_no), UART_RXFIFO_TOUT_INT_ENA | UART_RXFIFO_FULL_INT_ENA);
639  }
640 }
