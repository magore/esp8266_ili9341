80 {
81  uint32_t R;
82  switch (addr)
83  {
84  case 0 :
85  R = 0UL
86  | ((regs.r0.FRAC & 0x0fffUL ) << 3 )
87  | ((regs.r0.INT & 0xffffUL ) << 15 );
88  return(R );
89  case 1 :
90  R = 1UL
91  | (( regs.r1.MOD & 0x0fffUL ) << 3 )
92  | (( regs.r1.Phase & 0x0fffUL ) << 15 )
93  | (( regs.r1.Prescaler & 1UL ) << 27 )
94  | (( regs.r1.PhaseAdjust & 1UL ) << 28 );
95  return(R );
96  case 2 :
97  R = 2UL
98  | (( regs.r2.Reset_R_N & 1UL ) << 3 )
99  | (( regs.r2.CP3S & 1UL ) << 4 )
100  | (( regs.r2.PowerDown & 1UL ) << 5 )
101  | (( regs.r2.PhasePolarity & 1UL ) << 6 )
102  | (( regs.r2.LDP & 1UL ) << 7 )
103  | (( regs.r2.LDF & 1UL ) << 8 )
104  | (( regs.r2.CPC & 0xfUL ) << 9 )
105  | (( regs.r2.DBufRFDiv & 1UL ) << 13 )
106  | (( regs.r2.R & 0x3ffUL ) << 14 )
107  | (( regs.r2.REFinDIV2 & 1UL ) << 24 )
108  | (( regs.r2.REFinMUL2 & 1UL ) << 25 )
109  | (( regs.r2.MuxOut & 7UL ) << 26 )
110  | (( regs.r2.NoiseSpurMode & 3UL ) << 29 );
111  return(R );
112  case 3 :
113  R = 3UL
114  | (( regs.r3.ClkDiv & 0x0fffUL ) << 3 )
115  | (( regs.r3.ClkDivMode & 3UL ) << 15 )
116  | (( regs.r3.CSR & 1UL ) << 18 )
117  | (( regs.r3.CPC & 1UL ) << 21 )
118  | (( regs.r3.ABPW & 1UL ) << 22 )
119  | (( regs.r3.BandClkMode & 1UL ) << 23 );
120  return(R );
121  case 4 :
122  R = 4UL
123  | (( regs.r4.RFOutPower & 3UL ) << 3 )
124  | (( regs.r4.RFOutEnable & 1UL ) << 5 )
125  | (( regs.r4.AuxOutPower & 3UL ) << 6 )
126  | (( regs.r4.AuxOutEnable & 1UL ) << 8 )
127  | (( regs.r4.AuxOutSel & 1UL ) << 9 )

131  | (( regs.r4.RFDivSel & 7UL ) << 20 )
132  | (( regs.r4.FeedbackVCO & 1UL ) << 23 );
133  return(R );
134  case 5 :
135  R = 5UL
136  | (( 3UL ) << 19 ) // reserved high
137  | (( regs.r5.LDPinMode & 3UL ) << 22 );
138  return(R);
139  }

355 // PFD = REFin [(1 + REfinMUL2)/(R (1 + REFinDIV2))]
356 // REFin is the reference frequency input.
357 // REFinMUL2 is the REFin doubler bit (0 or 1).
358 // R is the RF reference division factor (1 to 1023).
359 // REFinDIV2 is the reference divide-by-2 bit (0 or 1).
360 MEMSPACE
361 double ADF4351_PFD(double REFin, int R)
362 {
363  double PFD = (uint32_t) REFin
364  * (regs.r2.REFinMUL2 ? 2 : 1)
365  / (R * (regs.r2.REFinDIV2 ? 2 : 1) );
366  return(PFD);
367 }

427 int ADF4351_Config(double RFout, double REFin, double ChannelSpacing, double *RFoutCalc )
428 {
429 
430  uint32_t div_gcd;
431  uint32_t r0_INT;
432  uint32_t r0_FRAC;
433  uint32_t r1_MOD;
434  uint32_t r1_Prescaler;
435  uint32_t r2_R;
436  uint32_t r4_RFDivSel;
437  uint32_t r4_BandClkDiv;
438  uint32_t RFoutDIV;
439  double N;
440  uint32_t N_min;
441 
442  double PFD;
443  double BandSelectClockFrequency;
444  double Fres;
445 
446  double dscale;
447  uint32_t temp;
448 
449  *RFoutCalc = 0.0;
450 
468  // ==========================
469  if (RFout > ADF4351_RFOUT_MAX)
470  {
471 #if ADF4351_DEBUG & 1
472  printf("RFout > %.2f\n", (double) ADF4351_RFOUT_MAX);
473 #endif
474  return(ADF4351_RFout_RANGE);
475  }
476 
477  if (RFout < ADF4351_RFOUT_MIN)
478  {
479 #if ADF4351_DEBUG & 1
480  printf("RFout < %.2f\n", (double) ADF4351_RFOUT_MIN);
481 #endif
482  return(ADF4351_RFout_RANGE);
483  }
484 
485  if (REFin > ADF4351_REFIN_MAX)
486  {
487 #if ADF4351_DEBUG & 1
488  printf("REFin > %.2f\n", (double) ADF4351_REFIN_MAX);
489 #endif
490  return(ADF4351_REFin_RANGE);
491  }
492 
493  // ==========================
494  // Compute RFout divider and R4 register value
495  r4_RFDivSel = 0;
496  RFoutDIV = 1;
497  while(((RFout * RFoutDIV) < ADF4351_VCO_MIN) && RFoutDIV < 64)
498  {
499  RFoutDIV <<= 1;
500  r4_RFDivSel++;
501  }
502 
503  // Compute r1_prescale selector based on RFout
504  if(RFout > ADF4351_MAX_FREQ_45PRE)
505  {
506  r1_Prescaler = 1;
507  N_min = 75;
508  }
509  else
510  {
511  r1_Prescaler = 0;
512  N_min = 23;
513  }
514 
515 
516  // ==========================
517  // PFD = REFin [(1 + REFinMUL2)/(R (1 + REFinDIV2))]
518  // REFin is the reference frequency input.
519  // REFinMUL2 is the REFin doubler bit (0 or 1).
520  // r2_R is REFin division factor (1 to 1023).
521  // REFinDIV2 is the reference divide-by-2 bit (0 or 1).
522  r2_R = 1;
523  while(r2_R < 4096 )
524  {
525  PFD = ADF4351_PFD(REFin, r2_R);
526  if(PFD < ADF4351_PFD_MAX)
527  break;
528  r2_R++;
529  }
530 
531  if(r2_R == 4096)
532  {
533 #if ADF4351_DEBUG & 1
534  printf("r2_R == 4096\n");
535 #endif
536  return(ADF4351_R_RANGE);
537  }
538 
539  // ==========================
540  // Compute N based on R4 feedback path select
541  if (regs.r4.FeedbackVCO)
542  N = ((double)RFout * (double)RFoutDIV) / (double) PFD;
543  else
544  N = ((double)RFout / (double) PFD);
545 
546  if(N < N_min || N > 65535U )
547  {
548 #if ADF4351_DEBUG & 1
549  printf("N %.2f out of range\n", (double) N);
550 #endif
551  return(ADF4351_N_RANGE);
552  }
553 
554  // ==========================
555  // Integer
556  r0_INT = (uint32_t) N;
557 
558  // r0_INt range check
559  if (r0_INT > 65535U)
560  {
561 #if ADF4351_DEBUG & 1
562  printf("INT: %lu\n", (unsigned long) r0_INT);
563 #endif
564  return(ADF4351_INT_RANGE);
565  }
566 
567  // ==========================
568  // Modulus
569  // FIXME the AD windows driver Main_Form.cs disagrees with their own datasheet
570  // Specifically the RF output divider is not included in the calculation
571  // Datasheet:
572  // Fres is the VCO Channel Spacing
573  // r1_MOD=REFin/Fres
574  // Fres = ChannelSpacing/RFoutDIV
575  // r1_MOD=(uint32_t) round((double) REFin / (double) Fres);
576 
577  // The tested working solution found in the AD driver Main_Form.cs
578  r1_MOD=(uint32_t) round((double) PFD / (double) ChannelSpacing );
579 
580  // ==========================
581  // Fractional
582  r0_FRAC = (uint32_t)round( ((double)N-(double)r0_INT)*(double) r1_MOD);
583 
584  // ==========================
585  // Reduce r1_MOD and r0_FRAC by greatest common divisor
586 
587  div_gcd = ADF4351_GCD32(r1_MOD, r0_FRAC);
588  r1_MOD /= div_gcd;
589  r0_FRAC /= div_gcd;
590 
591  // FIXME - why is this needed ?
592  if (r1_MOD == 1)
593  r1_MOD = 2;
594 
595  // ==========================
596  // r1_MOD Range check
597  if(r1_MOD == 0 || r1_MOD > 4095U)
598  {
599 #if ADF4351_DEBUG & 1
600  if(r1_MOD == 0)
601  printf("*MOD == 0\n");
602  if(r1_MOD > 4095U)
603  printf("*MOD > 4095\n");
604  printf("*MOD: %lu, INT: %lu, FRAC: %lu\n", (unsigned long) r1_MOD, (unsigned long) r0_INT, (unsigned long) r0_FRAC);
605 #endif
606  return(ADF4351_MOD_RANGE);
607  }
608 
609  // ==========================
610  // r0_FRAC range check
611  if (r0_FRAC > 4095U)
612  {
613 #if ADF4351_DEBUG & 1
614  if(r0_FRAC > 4095U)
615  printf("*FRAC > 4095\n");
616  printf("MOD: %lu, INT: %lu, *FRAC: %lu\n", (unsigned long) r1_MOD, (unsigned long) r0_INT, (unsigned long) r0_FRAC);
617 #endif
618  return(ADF4351_FRAC_RANGE);
619  }
620 
621  // ==========================
622  // Check for PFD range errors
623  if (regs.r3.BandClkMode == 0)
624  {
625  if( PFD > ADF4351_PFD_MAX )
626  {
627 #if ADF4351_DEBUG & 1
628  printf("PFD: %.2f > %lu && BandClkMode == 0\n",
629  (double) PFD, (unsigned long) ADF4351_PFD_MAX);
630 #endif
631  return(ADF4351_PFD_RANGE);
632  }
633  }
634  else
635  {
636  if( PFD > ADF4351_PFD_MAX && r0_FRAC != 0)
637  {
638 #if ADF4351_DEBUG & 1
639  printf("PFD: %.2f > %lu && BandClkMode == 0\n",
640  (double) PFD, (unsigned long) ADF4351_PFD_MAX);
641 #endif
642  return(ADF4351_PFD_RANGE);
643  }
644  if (PFD > 90 && r0_FRAC != 0)
645  {
646 #if ADF4351_DEBUG & 1
647  printf("PFD: %.2f > 90 Band Clock Mode && r0_FRAC != 0\n",
648  (double) PFD);
649 #endif
650  return(ADF4351_PFD_RANGE);
651  }
652  }
653 
654  // ==========================
655  // Band Clock Divider
656  // FIXME Add user override to pick value
657  // FIXME perhaps we could consider setting regs.r3.BandClkMode based on PFD ?
658  if (regs.r3.BandClkMode == 0)
659  dscale = 8;
660  else
661  dscale = 2;
662  temp = (uint32_t)round(dscale * PFD);
663  if ((dscale * PFD - temp) > 0)
664  temp++;
665 
666  temp = (temp > 255) ? 255 : temp;
667  r4_BandClkDiv = temp;
668 
669  // ==========================
670  // Band Clock Frequency
671  // FYI temp will always be > 0
672  BandSelectClockFrequency = (PFD / (double)r4_BandClkDiv);
673 
674  // ==========================
675  // Band Clock Range Check
676  if (BandSelectClockFrequency > 500000.0)
677  {
678 #if ADF4351_DEBUG & 1
679  printf("Band Select Clock Frequency %.2f > 500000\n",
680  (double) BandSelectClockFrequency);
681 #endif
682  return(ADF4351_BandSelectClockFrequency_RANGE);
683  }
684 
685  if ((BandSelectClockFrequency > 125000.0) & (regs.r3.BandClkMode))
686  {
687 #if ADF4351_DEBUG & 1
688  printf("Band Select Clock Frequency %.2f > 125000 && regs.r3.BandClkMode\n",
689  (double) BandSelectClockFrequency);
690 #endif
691  return(ADF4351_BandSelectClockFrequency_RANGE);
692  }
693
